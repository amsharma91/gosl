    type Bspline

    type Bspline struct {

    // essential
    T []float64 // array of knots: e.g: T = [t_0, t_1, ..., t_{m-1}]

    // optional
    Q [][]float64 // control points (has to call SetControl to change this)
    // contains filtered or unexported fields
}

    type NurbsD

    type NurbsD struct {
