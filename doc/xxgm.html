    type BezierQuad

    type BezierQuad struct {

    // input
    Q [][]float64 // control points; can be set outside

    // auxiliary
    P []float64 // a point on curve
}

    BezierQuad implements a quadratic Bezier curve 

    C(t) = (1-t)² Q0  +  2 t (1-t) Q1  +  t² Q2
     = t² A  +  2 t B  +  Q0
A = Q2 - 2 Q1 + Q0
B = Q1 - Q0
